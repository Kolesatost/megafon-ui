enum TargetType = {
    SELF: '_self',
    BLANK: '_blank',
};

interface IButton {
     /** Текст кнопки */
    title: string;
    /** Ссылка в кнопке */
    href: string;
    /** Target свойство, аналогично свойству 'target' тега 'a' */
    target: TargetType;
    /** Обработчик клика по кнопке */
    onClick: () => void;
}

interface ILink {
    /** Текст ссылки */
    title: string;
    /** Адрес ссылки */
    href?: string;
    /** Target свойство, аналогично свойству 'target' тега 'a' */
    target: TargetType;
}
