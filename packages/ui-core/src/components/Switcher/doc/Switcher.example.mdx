
import { Playground } from 'docz';
import Switcher from '../Switcher';
import { SwitcherWrapper } from './Switcher.docz.tsx';

```javascript collapse=Код SwitcherWrapper
export const SwitcherWrapper: React.FC<ISwitcherWrapperProps> = props => {
    const [checked, setChecked] = React.useState<boolean>(false);

    const handleChange = () => {
        setChecked(!checked);
    };

    return (
        <div style={{ padding: '20px 15px' }}>
            {props.children({
                checked,
                onChange: handleChange,
            })}
        </div>
    );
};
```

## Базовое использование

<Playground>
    <SwitcherWrapper>
        {({ checked, onChange }) => <Switcher checked={checked} onChange={onChange} />}
    </SwitcherWrapper>
</Playground>

## Переключатель отключен

### Состояние - "выкл."

<Playground>

    </SwitcherWrapper>
</Playground>

### Состояние - "вкл."

<Playground>
    <SwitcherWrapper>
        {({ onChange }) => <Switcher checked disabled onChange={onChange} />}
    </SwitcherWrapper>
</Playground>
